Com o comando abaixo é possível gerar um arquivo binário contendo os bytes de 0x00 a 0xff

[sourcecode language="shell"]root@M4v3r1ck:~# for i in {0..255}; do hex=$(printf %1x "$i"); echo -e -n "\x$hex"; done > bytearray.bin
[/sourcecode]
